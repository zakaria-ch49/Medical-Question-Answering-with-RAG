# ─── CI-only lightweight dependencies ────────────────────────────────────────
# These are the ONLY packages needed to run the unit-test suite on GitHub
# Actions.  Heavy ML packages (torch, sentence-transformers, faiss-cpu, …)
# are excluded because:

#      installing pytest (the root cause of "No module named pytest").
#   2. All code paths that use them are either lazy-loaded or mocked in tests.

pytest
pytest-mock

# ─── HTTP & env ──────────────────────────────────────────────────────────────
requests
python-dotenv
